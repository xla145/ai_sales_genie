下面是最终方案：Python Runner + 单 Hermes Agent 服务 + Project/Session/Run 模型 + 文件索引 + Patch 修改。

1. 核心设计

Project = 一个用户项目
Session = 这个项目对应的 Hermes 上下文
Run     = 一次生成或修改
Current = 项目当前稳定文件
Patch   = 每次修改的差异

目录结构：

data/
  projects/
    project_123/
      session.json
      memory.md
      project_index.json
      current/
        src/
        package.json
      runs/
        run_001/
          input.json
          events.jsonl
          patch.diff
          result.json

原则：

不把整个项目传给 Hermes
不传完整历史 messages
只传 memory + project_index + 相关文件
Hermes 输出 patch
Python Runner 负责 apply patch、测试、更新 current

5. Hermes Client

# hermes_client.py
import httpx

class HermesClient:
    def __init__(self, base_url: str, api_key: str | None = None):
        self.base_url = base_url.rstrip("/")
        self.api_key = api_key

    async def chat(self, messages: list[dict], temperature: float = 0.2) -> str:
        headers = {}
        if self.api_key:
            headers["Authorization"] = f"Bearer {self.api_key}"

        async with httpx.AsyncClient(timeout=300) as client:
            resp = await client.post(
                f"{self.base_url}/v1/chat/completions",
                headers=headers,
                json={
                    "model": "hermes",
                    "messages": messages,
                    "temperature": temperature,
                },
            )
            resp.raise_for_status()
            data = resp.json()
            return data["choices"][0]["message"]["content"]

6. Storage 层

# storage.py
import json
import shutil
from pathlib import Path
from uuid import uuid4

ROOT = Path("data/projects")

def create_project_dirs() -> tuple[str, Path]:
    project_id = f"project_{uuid4().hex[:12]}"
    root = ROOT / project_id
    (root / "current").mkdir(parents=True)
    (root / "runs").mkdir()
    (root / "memory.md").write_text("# Project Memory\n", encoding="utf-8")
    (root / "project_index.json").write_text(
        json.dumps({"framework": None, "entrypoints": [], "files": []}, indent=2),
        encoding="utf-8",
    )
    (root / "session.json").write_text(
        json.dumps({"messages": []}, indent=2),
        encoding="utf-8",
    )
    return project_id, root

def new_run_dir(project_id: str) -> tuple[str, Path]:
    run_id = f"run_{uuid4().hex[:12]}"
    path = ROOT / project_id / "runs" / run_id
    path.mkdir(parents=True)
    return run_id, path

def project_root(project_id: str) -> Path:
    return ROOT / project_id

def read_json(path: Path):
    return json.loads(path.read_text(encoding="utf-8"))

def write_json(path: Path, data):
    path.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")

def read_file(project_id: str, rel_path: str) -> str:
    path = project_root(project_id) / "current" / rel_path
    return path.read_text(encoding="utf-8")

def write_event(run_dir: Path, event: dict):
    with (run_dir / "events.jsonl").open("a", encoding="utf-8") as f:
        f.write(json.dumps(event, ensure_ascii=False) + "\n")

8. 第二步：只注入相关文件，让 Hermes 输出 Patch

# runner.py
import json
from pathlib import Path
from storage import project_root, read_file, write_event
from planner import plan_files
from patcher import apply_patch
from indexer import update_index_after_run

async def run_modify_job(hermes, project_id: str, run_id: str, run_dir: Path, user_message: str):
    write_event(run_dir, {"type": "status", "message": "开始分析相关文件"})

    plan = await plan_files(hermes, project_id, user_message)

    write_event(run_dir, {
        "type": "plan",
        "read": plan.get("read", []),
        "modify": plan.get("modify", [])
    })

    root = project_root(project_id)
    memory = (root / "memory.md").read_text(encoding="utf-8")
    index = (root / "project_index.json").read_text(encoding="utf-8")

    related_paths = sorted(set(plan.get("read", []) + plan.get("modify", [])))

    file_blocks = []
    for path in related_paths:
        try:
            content = read_file(project_id, path)
            file_blocks.append(f"--- FILE: {path} ---\n{content}")
        except FileNotFoundError:
            file_blocks.append(f"--- FILE: {path} ---\n<FILE_NOT_FOUND>")

    messages = [
        {
            "role": "system",
            "content": """
你是代码修改 Agent。
你必须基于用户需求修改项目。
要求：
1. 只输出 unified diff patch。
2. 不要输出解释。
3. 不要输出完整文件。
4. patch 路径必须相对于 current 目录。
5. 如果需要新增文件，也用 unified diff 表示。
"""
        },
        {
            "role": "user",
            "content": f"""
项目记忆：
{memory}

项目索引：
{index}

用户需求：
{user_message}

相关文件：
{chr(10).join(file_blocks)}
"""
        }
    ]

    write_event(run_dir, {"type": "status", "message": "正在生成 patch"})

    patch = await hermes.chat(messages)
    (run_dir / "patch.diff").write_text(patch, encoding="utf-8")

    write_event(run_dir, {"type": "patch_created", "path": "patch.diff"})

    apply_patch(root / "current", run_dir / "patch.diff")

    write_event(run_dir, {"type": "status", "message": "patch 已应用"})

    await update_index_after_run(hermes, project_id, related_paths, user_message)

    write_event(run_dir, {"type": "done", "message": "修改完成"})

9. Patch 应用

# patcher.py
import subprocess
from pathlib import Path

def apply_patch(current_dir: Path, patch_file: Path):
    result = subprocess.run(
        ["git", "apply", str(patch_file)],
        cwd=current_dir,
        text=True,
        capture_output=True,
    )

    if result.returncode != 0:
        raise RuntimeError(
            f"apply patch failed\nSTDOUT:\n{result.stdout}\nSTDERR:\n{result.stderr}"
        )

建议 current/ 初始化成 git repo：

10. 更新 memory 和 index

# indexer.py
import json
from storage import project_root, read_json, write_json, read_file

async def update_index_after_run(hermes, project_id: str, changed_paths: list[str], user_message: str):
    root = project_root(project_id)
    memory_path = root / "memory.md"
    index_path = root / "project_index.json"

    memory = memory_path.read_text(encoding="utf-8")
    index = read_json(index_path)

    changed_content = []
    for path in changed_paths:
        try:
            changed_content.append(f"--- FILE: {path} ---\n{read_file(project_id, path)}")
        except Exception:
            pass

    messages = [
        {
            "role": "system",
            "content": """
你负责维护项目长期记忆和文件索引。
只返回 JSON：
{
  "memory": "新的 memory.md 内容",
  "files": [
    {"path": "src/App.tsx", "summary": "文件摘要"}
  ]
}
"""
        },
        {
            "role": "user",
            "content": f"""
旧 memory：
{memory}

旧 index：
{json.dumps(index, ensure_ascii=False)}

本次用户需求：
{user_message}

变更文件：
{chr(10).join(changed_content)}
"""
        }
    ]

    raw = await hermes.chat(messages)
    data = json.loads(raw)

    memory_path.write_text(data["memory"], encoding="utf-8")

    existing = {f["path"]: f for f in index.get("files", [])}
    for f in data.get("files", []):
        existing[f["path"]] = f

    index["files"] = list(existing.values())
    write_json(index_path, index)

11. FastAPI 主服务

# main.py
import asyncio
from fastapi import FastAPI, HTTPException
from fastapi.responses import StreamingResponse
from models import CreateProjectRequest, RunRequest
from storage import create_project_dirs, new_run_dir, project_root, write_json
from hermes_client import HermesClient
from runner import run_modify_job

app = FastAPI()

hermes = HermesClient(
    base_url="http://localhost:8000",
    api_key=None,
)

job_locks = {}
project_locks = {}

@app.post("/projects")
async def create_project(req: CreateProjectRequest):
    project_id, root = create_project_dirs()
    run_id, run_dir = new_run_dir(project_id)

    write_json(run_dir / "input.json", {
        "type": "create",
        "prompt": req.prompt,
    })

    async def task():
        await run_modify_job(
            hermes=hermes,
            project_id=project_id,
            run_id=run_id,
            run_dir=run_dir,
            user_message=req.prompt,
        )

    asyncio.create_task(task())

    return {
        "project_id": project_id,
        "run_id": run_id,
        "events_url": f"/projects/{project_id}/runs/{run_id}/events",
    }

@app.post("/projects/{project_id}/runs")
async def create_run(project_id: str, req: RunRequest):
    root = project_root(project_id)
    if not root.exists():
        raise HTTPException(404, "project not found")

    run_id, run_dir = new_run_dir(project_id)

    if project_id not in project_locks:
        project_locks[project_id] = asyncio.Lock()

    async def task():
        async with project_locks[project_id]:
            await run_modify_job(
                hermes=hermes,
                project_id=project_id,
                run_id=run_id,
                run_dir=run_dir,
                user_message=req.message,
            )

    asyncio.create_task(task())

    return {
        "project_id": project_id,
        "run_id": run_id,
        "events_url": f"/projects/{project_id}/runs/{run_id}/events",
    }

@app.get("/projects/{project_id}/runs/{run_id}/events")
async def stream_events(project_id: str, run_id: str):
    run_dir = project_root(project_id) / "runs" / run_id
    event_file = run_dir / "events.jsonl"

    async def generator():
        pos = 0
        while True:
            if event_file.exists():
                with event_file.open("r", encoding="utf-8") as f:
                    f.seek(pos)
                    lines = f.readlines()
                    pos = f.tell()

                for line in lines:
                    yield f"data: {line.strip()}\n\n"
                    if '"type": "done"' in line or '"type": "error"' in line:
                        return

            await asyncio.sleep(0.5)

    return StreamingResponse(generator(), media_type="text/event-stream")

13. 最推荐的执行策略

第一版：

单 Hermes 服务
全局并发 = 1
同一个 project 串行
每次 run 只传相关文件
每次 run 输出 patch
current 用 git 管版本

第二版再优化：

同一个 project 串行
不同 project 可并发
失败 patch 自动重试
接入 build/test
接入 zip 下载和 preview

一句话总结：Python Runner 管项目、文件、索引、日志和版本；Hermes 只负责规划相关文件和生成 patch。
